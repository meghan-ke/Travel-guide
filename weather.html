async loadCountryDetails(countryName) { const container = document.getElementById('countryDetailsContainer'); try { const countries = await Utils.fetchCountriesByName(countryName); if (!countries || countries.length === 0) { throw new Error('Country not found'); } const country = countries[0]; this.displayCountryDetails(country); } catch (error) { console.error('Error loading country details:', error); this.showCountryError('Country not found'); } }
